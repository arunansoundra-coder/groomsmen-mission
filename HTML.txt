// Scroll trigger for pages
const pages = document.querySelectorAll('.page');
window.addEventListener('scroll', () => {
  pages.forEach(page => {
    const rect = page.getBoundingClientRect();
    if(rect.top < window.innerHeight/1.2){
      page.classList.add('active');
    }
  });
});

// Typing effect for puzzle
const puzzleText = "ZLOO BRX EH PB JURRPVPDQ?";
const puzzleDiv = document.getElementById('puzzle');
let i=0;
function typePuzzle(){
  if(i<puzzleText.length){
    puzzleDiv.innerHTML += puzzleText.charAt(i);
    i++;
    setTimeout(typePuzzle, 100);
  }
}
typePuzzle();
</script>
